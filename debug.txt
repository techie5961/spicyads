       try{
        if(this.files[0]){
        document.querySelector('.prev-img').src=URL.createObjectURL(this.files[0]);

        document.querySelector('.prompt').classList.add('display-none');
        }else{
          document.querySelector('.prev-img').classList.add('display-none');

        }
